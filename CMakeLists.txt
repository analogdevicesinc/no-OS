function(no_os_sources_ifdef feature_toggle source)
  if(${feature_toggle})
    message(STATUS "Adding ${source}")
    target_sources(capi_build PRIVATE ${source} ${ARGN})
  endif()
endfunction()

function(capi_update_config)
  include(${CMAKE_SOURCE_DIR}/config.cmake)
endfunction()
