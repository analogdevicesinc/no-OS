56/* Communications Register Bit Designations (AD719X_REG_COMM) */
57#define AD719X_COMM_WEN (1 << 7) // Write Enable.
58#define AD719X_COMM_WRITE (0 << 6) // Write Operation.
59#define AD719X_COMM_READ (1 << 6) // Read Operation.
60#define AD719X_COMM_ADDR(x) (((x) & 0x7) << 3) // Register Address.
61#define AD719X_COMM_CREAD (1 << 2) // Continuous Read of Data Register.
62
63/* Status Register Bit Designations (AD719X_REG_STAT) */
64#define AD719X_STAT_RDY (1 << 7) // Ready.
65#define AD719X_STAT_ERR (1 << 6) // ADC error bit.
66#define AD719X_STAT_NOREF (1 << 5) // Error no external reference.
67#define AD719X_STAT_PARITY (1 << 4) // Parity check of the data register.
68#define AD719X_STAT_CH3 (1 << 3) // Channel 3.
69#define AD719X_STAT_CH2 (1 << 2) // Channel 2.
70#define AD719X_STAT_CH1 (1 << 1) // Channel 1.
71#define AD719X_STAT_CH0 (1 << 0) // Channel 0.
72
73/* Mode Register Bit Designations (AD719X_REG_MODE) */
74#define AD719X_MODE_SEL(x) (((uint32_t)(x) & 0x7) << 21) // Operation Mode Select.
75#define AD719X_MODE_DAT_STA ((uint32_t)1 << 20) // Status Register transmission.
76#define AD719X_MODE_CLKSRC(x) (((uint32_t)(x) & 0x3) << 18) // Clock Source Select.
77#define AD719X_MODE_AVG(x) (((uint32_t)(x) & 0x3) << 16) // Fast settling filter.
78#define AD719X_MODE_SINC3 (1 << 15) // SINC3 Filter Select.
79#define AD719X_MODE_ENPAR (1 << 13) // Parity Enable.
80#define AD719X_MODE_CLKDIV (1 << 12) // Clock divide by 2 (AD7190/2 only).
81#define AD719X_MODE_SCYCLE (1 << 11) // Single cycle conversion.
82#define AD719X_MODE_REJ60 (1 << 10) // 50/60Hz notch filter.
83#define AD719X_MODE_RATE(x) ((x) & 0x3FF) // Filter Update Rate Select.
84
85/* Mode Register: AD719X_MODE_AVG(x) options */
86#define AD719X_AVG_NONE 0 // No averaging (fast settling mode disabled).
87#define AD719X_AVG_BY_2 1 // Average by 2.
88#define AD719X_AVG_BY_8 2 // Average by 8.
89#define AD719X_AVG_BY_16 3 // Average by 16.
90
91/* Configuration Register Bit Designations (AD719X_REG_CONF) */
92#define AD719X_CONF_CHOP ((uint32_t)1 << 23) // CHOP enable.
93#define AD719X_CONF_REFSEL ((uint32_t)1 << 20) // REFIN1/REFIN2 Reference Select.
94#define AD719X_CONF_PSEUDO ((uint32_t)1 << 18) // Pseudo differential analog inputs.
95#define AD719X_CONF_CHAN(x) ((uint32_t)((x) & 0x3FF) << 8) // Channel select.
96#define AD719X_CONF_BURN (1 << 7) // Burnout current enable.
97#define AD719X_CONF_REFDET (1 << 6) // Reference detect enable.
98#define AD719X_CONF_BUF (1 << 4) // Buffered Mode Enable.
99#define AD719X_CONF_UNIPOLAR (1 << 3) // Unipolar/Bipolar Enable.
100#define AD719X_CONF_GAIN(x) ((x) & 0x7) // Gain Select.
101
